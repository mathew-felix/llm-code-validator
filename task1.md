def generate_report_node(state: AgentState) -> AgentState:
    raw_issues = state.get("diagnosed_issues", [])

    # Filter out low-confidence flags — they are almost always false positives
    CONFIDENCE_THRESHOLD = 0.75
    confident_issues = [
        issue for issue in raw_issues
        if issue.confidence >= CONFIDENCE_THRESHOLD
    ]

    report = ValidationReport(
        issues=confident_issues,
        total_issues=len(confident_issues),
        libraries_checked=state.get("libraries_checked", []),
        filtered_count=len(raw_issues) - len(confident_issues)
    )

    return {**state, "report": report}

    code = state["code"]
    imports = state["imports"]

    with open(DATABASE_PATH) as f:
        database = json.load(f)

    database_matches = []

    for imp in imports:
        library_name = imp["library"].split(".")[0]  # top-level package only

        if library_name not in database:
            libraries_to_fetch.append(library_name)
            continue

        library_data = database[library_name]

        for method_name, method_data in library_data.get("methods", {}).items():
            # PRE-FILTER: only include this entry if the method name
            # or the old import path actually appears in the code
            # This prevents the LLM from reasoning about methods never called
            method_appears_in_code = (
                method_name in code or
                method_data.get("old_import", "") in code or
                any(call["method"] == method_name for call in method_calls)
            )

            if not method_appears_in_code:
                continue  # skip — LLM never sees this entry

            database_matches.append({
                "library": library_name,
                "method": method_name,
                "data": method_data
            })

    needs_pypi_fetch = len(libraries_to_fetch) > 0

    return {
        **state,
        "database_matches": database_matches,
        "libraries_to_fetch": libraries_to_fetch,
        "needs_pypi_fetch": needs_pypi_fetch
    }

from agent.schemas import AgentState

DATABASE_PATH = os.path.join(os.path.dirname(__file__),

IMPORT_SPECIALIST_PROMPT = """You are an import path validation specialist.
Your ONLY job: determine if any import statements use old module paths
that were restructured in a breaking version change.

You receive:
- import_database_matches: entries where the import PATH changed (not the method)
- raw_imports: the exact import statements from the code

Rules:
- Only flag issues where the import MODULE PATH is wrong
- Example: "from langchain.chat_models import ChatOpenAI" is wrong —
  should be "from langchain_openai import ChatOpenAI"
- Do NOT flag method-level issues — that is the Method Specialist's job
- Assign confidence scores as instructed

Return ONLY import path issues. Nothing else."""

def import_specialist_node(state: AgentState) -> AgentState:
    """
    Specialist agent focused exclusively on wrong import module paths.
    Catches cases where library restructuring changed where things live.
    """
    # Filter database matches to only import-level changes
    import_matches = [
        m for m in state.get("database_matches", [])
        if m["data"].get("old_import") and m["data"].get("new_import")
    ]

    if not import_matches:
        return {**state, "import_specialist_findings": []}

    # Build specialist-specific prompt context
    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    context = {
        "imports_in_code": state.get("imports", []),
        "import_database_matches": import_matches
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.1,
        messages=[
            {"role": "system", "content": IMPORT_SPECIALIST_PROMPT},
            {"role": "user", "content": f"Validate these imports:\n{json.dumps(context, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    findings = json.loads(response.choices[0].message.content)
    return {**state, "import_specialist_findings": findings.get("issues", [])}

from agent.schemas import AgentState
import json

SUPERVISOR_PROMPT = """You are a code validation supervisor.
Your job: read a Python code snippet and decide which specialist
validation agents need to run.

Available specialists:
- "import_specialist": detects wrong import module paths (restructured packages)
- "method_specialist": detects deprecated method calls (removed methods)
- "version_specialist": detects version-specific API changes

Based on what you see in the code, return a JSON with:
{
  "specialists_needed": ["import_specialist", "method_specialist"],
  "reasoning": "Code uses langchain and pandas — both have known restructuring"
}

Only include specialists that are relevant to what is actually in the code.
If the code has no imports from known libraries, return an empty list."""

def supervisor_node(state: AgentState) -> AgentState:
    """
    Supervisor agent that reads the code and decides which specialist
    agents to invoke. Uses LLM reasoning, not deterministic conditionals.
    This is the key architectural difference from a simple pipeline.
    """
    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    code_summary = {
        "imports": state.get("imports", []),
        "method_calls": state.get("method_calls", []),
        "libraries_detected": list(set(
            imp["library"].split(".")[0]
            for imp in state.get("imports", [])
        ))
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.0,  # deterministic routing decisions
        messages=[
            {"role": "system", "content": SUPERVISOR_PROMPT},
            {"role": "user", "content": f"Code analysis:\n{json.dumps(code_summary, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    decision = json.loads(response.choices[0].message.content)

    return {
        **state,
        "specialists_needed": decision.get("specialists_needed", []),
        "supervisor_reasoning": decision.get("reasoning", "")
    }

from langgraph.graph import StateGraph, END
from agent.schemas import AgentState
from agent.nodes.extract_imports import extract_imports_node
from agent.nodes.check_database import check_database_node
from agent.nodes.fetch_pypi import fetch_pypi_node
from agent.nodes.supervisor import supervisor_node
from agent.nodes.import_specialist import import_specialist_node
from agent.nodes.method_specialist import method_specialist_node
from agent.nodes.generate_report import generate_report_node

def route_to_pypi(state: AgentState) -> str:
    """Route to PyPI fetch if unknown libraries detected."""
    return "fetch_pypi" if state.get("needs_pypi_fetch") else "supervisor"

def route_specialists(state: AgentState) -> list[str]:
    """
    Supervisor routing — returns list of specialist nodes to invoke.
    This is LLM-driven routing, not a hardcoded conditional.
    """
    needed = state.get("specialists_needed", [])
    routes = []

    if "import_specialist" in needed:
        routes.append("import_specialist")
    if "method_specialist" in needed:
        routes.append("method_specialist")

    # Always run at least one specialist
    if not routes:
        routes.append("import_specialist")

    return routes

def build_graph() -> StateGraph:
    graph = StateGraph(AgentState)

    # Add all nodes
    graph.add_node("extract_imports", extract_imports_node)
    graph.add_node("check_database", check_database_node)
    graph.add_node("fetch_pypi", fetch_pypi_node)
    graph.add_node("supervisor", supervisor_node)
    graph.add_node("import_specialist", import_specialist_node)
    graph.add_node("method_specialist", method_specialist_node)
    graph.add_node("generate_report", generate_report_node)

    # Set entry point
    graph.set_entry_point("extract_imports")

    # Linear flow: extract → database check
    graph.add_edge("extract_imports", "check_database")

    # Conditional: known libraries → supervisor | unknown → PyPI first
    graph.add_conditional_edges(
        "check_database",
        route_to_pypi,
        {
            "fetch_pypi": "fetch_pypi",
            "supervisor": "supervisor"
        }
    )

    # PyPI fetch always flows to supervisor
    graph.add_edge("fetch_pypi", "supervisor")

    # Supervisor routes to specialists (LLM-driven routing)
    graph.add_conditional_edges(
        "supervisor",
        route_specialists,
        {
            "import_specialist": "import_specialist",
            "method_specialist": "method_specialist"
        }
    )

    # Both specialists flow to report generation
    graph.add_edge("import_specialist", "generate_report")
    graph.add_edge("method_specialist", "generate_report")

    # Report generation ends the graph
    graph.add_edge("generate_report", END)

    return graph.compile()

# Build the compiled graph — imported by api/main.py
compiled_graph = build_graph()

from agent.schemas import AgentState, ValidationReport, ValidationIssue

CONFIDENCE_THRESHOLD = 0.75

def generate_report_node(state: AgentState) -> AgentState:
    """
    Aggregates findings from all specialist agents.
    Deduplicates, applies confidence threshold, builds final report.
    """
    # Collect all findings from all specialists
    all_findings = []
    all_findings.extend(state.get("import_specialist_findings", []))
    all_findings.extend(state.get("method_specialist_findings", []))

    # Parse into ValidationIssue objects if they are dicts
    parsed_issues = []
    for finding in all_findings:
        if isinstance(finding, dict):
            try:
                parsed_issues.append(ValidationIssue(**finding))
            except Exception:
                continue
        elif isinstance(finding, ValidationIssue):
            parsed_issues.append(finding)

    # Apply confidence threshold
    confident_issues = [
        issue for issue in parsed_issues
        if issue.confidence >= CONFIDENCE_THRESHOLD
    ]

    # Deduplicate: same library + method = same issue
    seen = set()
    deduplicated = []
    for issue in confident_issues:
        key = f"{issue.library}:{issue.method}:{issue.line_number}"
        if key not in seen:
            seen.add(key)
            deduplicated.append(issue)

    report = ValidationReport(
        issues=deduplicated,
        total_issues=len(deduplicated),
        libraries_checked=state.get("libraries_checked", []),
        filtered_count=len(parsed_issues) - len(deduplicated)
    )

    return {**state, "report": report}

Time: 30 minutes | File: agent/nodes/check_database.py
What this task is:
Before sending anything to the LLM, check whether the deprecated method

Why this task is needed:
Your current flow sends ALL database matches to the LLM for every import

How it helps the problem:

def check_database_node(state: AgentState) -> AgentState:
    code = state["code"]
    imports = state["imports"]
    method_calls = state.get("method_calls", [])

    with open(DATABASE_PATH) as f:
        database = json.load(f)

    database_matches = []
    libraries_to_fetch = []

    for imp in imports:
        library_name = imp["library"].split(".")[0]  # top-level package only

        if library_name not in database:
            libraries_to_fetch.append(library_name)
            continue

        library_data = database[library_name]

        for method_name, method_data in library_data.get("methods", {}).items():
            # PRE-FILTER: only include this entry if the method name
            # or the old import path actually appears in the code
            # This prevents the LLM from reasoning about methods never called
            method_appears_in_code = (
                method_name in code or
                method_data.get("old_import", "") in code or
                any(call["method"] == method_name for call in method_calls)
            )

            if not method_appears_in_code:
                continue  # skip — LLM never sees this entry

            database_matches.append({
                "library": library_name,
                "method": method_name,
                "data": method_data
            })

    needs_pypi_fetch = len(libraries_to_fetch) > 0

    return {
        **state,
        "database_matches": database_matches,
        "libraries_to_fetch": libraries_to_fetch,
        "needs_pypi_fetch": needs_pypi_fetch
    }

IMPORT_SPECIALIST_PROMPT = """You are an import path validation specialist.
Your ONLY job: determine if any import statements use old module paths
that were restructured in a breaking version change.

You receive:
- import_database_matches: entries where the import PATH changed (not the method)
- raw_imports: the exact import statements from the code

Rules:
- Only flag issues where the import MODULE PATH is wrong
- Example: "from langchain.chat_models import ChatOpenAI" is wrong —
  should be "from langchain_openai import ChatOpenAI"
- Do NOT flag method-level issues — that is the Method Specialist's job
- Assign confidence scores as instructed

Return ONLY import path issues. Nothing else."""

def import_specialist_node(state: AgentState) -> AgentState:
    """
    Specialist agent focused exclusively on wrong import module paths.
    Catches cases where library restructuring changed where things live.
    """
    # Filter database matches to only import-level changes
    import_matches = [
        m for m in state.get("database_matches", [])
        if m["data"].get("old_import") and m["data"].get("new_import")
    ]

    if not import_matches:
        return {**state, "import_specialist_findings": []}

    # Build specialist-specific prompt context
    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    context = {
        "imports_in_code": state.get("imports", []),
        "import_database_matches": import_matches
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.1,
        messages=[
            {"role": "system", "content": IMPORT_SPECIALIST_PROMPT},
            {"role": "user", "content": f"Validate these imports:\n{json.dumps(context, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    findings = json.loads(response.choices[0].message.content)
    return {**state, "import_specialist_findings": findings.get("issues", [])}

    graph.add_edge("method_specialist", "generate_report")

    Aggregates findings from all specialist agents.

    all_findings = []

    all_findings.extend(state.get("method_specialist_findings", []))

    # Parse into ValidationIssue objects if they are dicts
    parsed_issues = []
    for finding in all_findings:
        if isinstance(finding, dict):
            try:
                parsed_issues.append(ValidationIssue(**finding))
            except Exception:
                continue
        elif isinstance(finding, ValidationIssue):
            parsed_issues.append(finding)

    # Apply confidence threshold
    confident_issues = [
        issue for issue in parsed_issues
        if issue.confidence >= CONFIDENCE_THRESHOLD
    ]

    # Deduplicate: same library + method = same issue
    seen = set()
    deduplicated = []
    for issue in confident_issues:
        key = f"{issue.library}:{issue.method}:{issue.line_number}"
        if key not in seen:
            seen.add(key)
            deduplicated.append(issue)

    report = ValidationReport(
        issues=deduplicated,
        total_issues=len(deduplicated),
        libraries_checked=state.get("libraries_checked", []),
        filtered_count=len(parsed_issues) - len(deduplicated)
    )

    return {**state, "report": report}

## Task 1.1 — Add Confidence Scoring to ValidationIssue Schema

**What this task is:**

## Task 1.2 — Add Exact String Pre-Filter Before LLM
### Time: 30 minutes | File: agent/nodes/check_database.py

***

**What this task is:**
Before sending anything to the LLM, check whether the deprecated method
string is literally present anywhere in the raw code. If it is not there,
skip the LLM for that database entry entirely.

**Why this task is needed:**
Your current flow sends ALL database matches to the LLM for every import
it finds. This means if someone imports `pandas`, the LLM receives all 8+
pandas database entries and reasons about whether any of them apply —
even if the code never calls `df.append()` or any other deprecated method.
The LLM then sometimes hallucinates that a method is used when it is not.

**How it helps the problem:**
This pre-filter eliminates the LLM's ability to flag a deprecated method
that is not even called in the code. It is the cheapest precision fix
available — 30 lines of Python that remove an entire class of false positives.

**What to add in check_database.py:**

Find the section where you build `database_matches` to pass to the LLM.
Add a filter right before that list is assembled:

```python
def check_database_node(state: AgentState) -> AgentState:
    code = state["code"]
    imports = state["imports"]
    method_calls = state.get("method_calls", [])

    with open(DATABASE_PATH) as f:
        database = json.load(f)

    database_matches = []
    libraries_to_fetch = []

    for imp in imports:
        library_name = imp["library"].split(".")[0]  # top-level package only

        if library_name not in database:
            libraries_to_fetch.append(library_name)
            continue

        library_data = database[library_name]

        for method_name, method_data in library_data.get("methods", {}).items():
            # PRE-FILTER: only include this entry if the method name
            # or the old import path actually appears in the code
            # This prevents the LLM from reasoning about methods never called
            method_appears_in_code = (
                method_name in code or
                method_data.get("old_import", "") in code or
                any(call["method"] == method_name for call in method_calls)
            )

            if not method_appears_in_code:
                continue  # skip — LLM never sees this entry

            database_matches.append({
                "library": library_name,
                "method": method_name,
                "data": method_data
            })

    needs_pypi_fetch = len(libraries_to_fetch) > 0

    return {
        **state,
        "database_matches": database_matches,
        "libraries_to_fetch": libraries_to_fetch,
        "needs_pypi_fetch": needs_pypi_fetch
    }
```

## Task 3.1 — Understand What You Are Building
### Time: 20 minutes (reading only)

**What you are building:**
Right now you have a flat graph with 5 nodes and 1 conditional edge.
You are upgrading it to a supervisor-specialist multi-agent architecture.

**What that means:**
Instead of one graph that does everything, you will have:
- 1 **Supervisor agent** that reads the code and decides what to check
- 3 **Specialist agents** (subgraphs) that each focus on one problem type
- 1 **Aggregator** that combines their findings into the final report

**Why this is meaningfully harder than what you have:**
Your current conditional edge is deterministic — it checks a boolean flag.
A supervisor agent uses LLM reasoning to make routing decisions.
This is the pattern used in production systems at AWS, Google, and Anthropic.
The LangGraph documentation calls this the "supervisor pattern" and it is
explicitly listed as an advanced architecture beyond the quickstart tutorial.

**The new graph looks like this:**
```
[Input]
   ↓
[Supervisor Agent]
   ↓ routes based on what it detects in the code
   ├─→ [Import Specialist] ← checks wrong imports and module restructuring
   ├─→ [Method Specialist] ← checks deprecated method calls
   └─→ [Version Specialist] ← checks version-specific breakages
   ↓ all results return to supervisor
[Aggregator Node]
   ↓
[Report Generator]
```

```python
import json
import os
from agent.schemas import AgentState

DATABASE_PATH = os.path.join(os.path.dirname(__file__),
                             "../../data/library_signatures.json")

IMPORT_SPECIALIST_PROMPT = """You are an import path validation specialist.
Your ONLY job: determine if any import statements use old module paths
that were restructured in a breaking version change.

You receive:
- import_database_matches: entries where the import PATH changed (not the method)
- raw_imports: the exact import statements from the code

Rules:
- Only flag issues where the import MODULE PATH is wrong
- Example: "from langchain.chat_models import ChatOpenAI" is wrong —
  should be "from langchain_openai import ChatOpenAI"
- Do NOT flag method-level issues — that is the Method Specialist's job
- Assign confidence scores as instructed

Return ONLY import path issues. Nothing else."""

def import_specialist_node(state: AgentState) -> AgentState:
    """
    Specialist agent focused exclusively on wrong import module paths.
    Catches cases where library restructuring changed where things live.
    """
    # Filter database matches to only import-level changes
    import_matches = [
        m for m in state.get("database_matches", [])
        if m["data"].get("old_import") and m["data"].get("new_import")
    ]

    if not import_matches:
        return {**state, "import_specialist_findings": []}

    # Build specialist-specific prompt context
    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    context = {
        "imports_in_code": state.get("imports", []),
        "import_database_matches": import_matches
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.1,
        messages=[
            {"role": "system", "content": IMPORT_SPECIALIST_PROMPT},
            {"role": "user", "content": f"Validate these imports:\n{json.dumps(context, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    findings = json.loads(response.choices[0].message.content)
    return {**state, "import_specialist_findings": findings.get("issues", [])}
```

```python
from agent.schemas import AgentState
import json
import os

METHOD_SPECIALIST_PROMPT = """You are a deprecated method call specialist.
Your ONLY job: determine if any method CALLS in the code use methods that
have been removed or renamed in a breaking version change.

You receive:
- method_database_matches: database entries for removed/changed methods
- method_calls: all attribute method calls extracted from the code via AST

Rules:
- Only flag issues where a CALLED method is deprecated or removed
- The method must actually be CALLED in the code — not just imported
- Example: df.append() is a call. "from pandas import append" is an import.
- Assign confidence scores 0.0–1.0. Only return issues with confidence >= 0.75.

Return ONLY method-level deprecation issues."""

def method_specialist_node(state: AgentState) -> AgentState:
    """
    Specialist agent focused on deprecated method calls on objects.
    Catches df.append(), session.query(), pinecone.init() style failures.
    """
    method_calls = state.get("method_calls", [])
    method_matches = [
        m for m in state.get("database_matches", [])
        if m["data"].get("exists") == False  # methods that were removed
    ]

    if not method_calls or not method_matches:
        return {**state, "method_specialist_findings": []}

    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    context = {
        "method_calls_in_code": method_calls,
        "deprecated_method_entries": method_matches
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.1,
        messages=[
            {"role": "system", "content": METHOD_SPECIALIST_PROMPT},
            {"role": "user", "content": f"Check these method calls:\n{json.dumps(context, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    findings = json.loads(response.choices[0].message.content)
    return {**state, "method_specialist_findings": findings.get("issues", [])}
```

```python
from agent.schemas import AgentState
import json

SUPERVISOR_PROMPT = """You are a code validation supervisor.
Your job: read a Python code snippet and decide which specialist
validation agents need to run.

Available specialists:
- "import_specialist": detects wrong import module paths (restructured packages)
- "method_specialist": detects deprecated method calls (removed methods)
- "version_specialist": detects version-specific API changes

Based on what you see in the code, return a JSON with:
{
  "specialists_needed": ["import_specialist", "method_specialist"],
  "reasoning": "Code uses langchain and pandas — both have known restructuring"
}

Only include specialists that are relevant to what is actually in the code.
If the code has no imports from known libraries, return an empty list."""

def supervisor_node(state: AgentState) -> AgentState:
    """
    Supervisor agent that reads the code and decides which specialist
    agents to invoke. Uses LLM reasoning, not deterministic conditionals.
    This is the key architectural difference from a simple pipeline.
    """
    from openai import OpenAI
    client = OpenAI(timeout=30.0)

    code_summary = {
        "imports": state.get("imports", []),
        "method_calls": state.get("method_calls", []),
        "libraries_detected": list(set(
            imp["library"].split(".")[0]
            for imp in state.get("imports", [])
        ))
    }

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0.0,  # deterministic routing decisions
        messages=[
            {"role": "system", "content": SUPERVISOR_PROMPT},
            {"role": "user", "content": f"Code analysis:\n{json.dumps(code_summary, indent=2)}"}
        ],
        response_format={"type": "json_object"}
    )

    decision = json.loads(response.choices[0].message.content)

    return {
        **state,
        "specialists_needed": decision.get("specialists_needed", []),
        "supervisor_reasoning": decision.get("reasoning", "")
    }
```

```python
from langgraph.graph import StateGraph, END
from agent.schemas import AgentState
from agent.nodes.extract_imports import extract_imports_node
from agent.nodes.check_database import check_database_node
from agent.nodes.fetch_pypi import fetch_pypi_node
from agent.nodes.supervisor import supervisor_node
from agent.nodes.import_specialist import import_specialist_node
from agent.nodes.method_specialist import method_specialist_node
from agent.nodes.generate_report import generate_report_node

def route_to_pypi(state: AgentState) -> str:
    """Route to PyPI fetch if unknown libraries detected."""
    return "fetch_pypi" if state.get("needs_pypi_fetch") else "supervisor"

def route_specialists(state: AgentState) -> list[str]:
    """
    Supervisor routing — returns list of specialist nodes to invoke.
    This is LLM-driven routing, not a hardcoded conditional.
    """
    needed = state.get("specialists_needed", [])
    routes = []

    if "import_specialist" in needed:
        routes.append("import_specialist")
    if "method_specialist" in needed:
        routes.append("method_specialist")

    # Always run at least one specialist
    if not routes:
        routes.append("import_specialist")

    return routes

def build_graph() -> StateGraph:
    graph = StateGraph(AgentState)

    # Add all nodes
    graph.add_node("extract_imports", extract_imports_node)
    graph.add_node("check_database", check_database_node)
    graph.add_node("fetch_pypi", fetch_pypi_node)
    graph.add_node("supervisor", supervisor_node)
    graph.add_node("import_specialist", import_specialist_node)
    graph.add_node("method_specialist", method_specialist_node)
    graph.add_node("generate_report", generate_report_node)

    # Set entry point
    graph.set_entry_point("extract_imports")

    # Linear flow: extract → database check
    graph.add_edge("extract_imports", "check_database")

    # Conditional: known libraries → supervisor | unknown → PyPI first
    graph.add_conditional_edges(
        "check_database",
        route_to_pypi,
        {
            "fetch_pypi": "fetch_pypi",
            "supervisor": "supervisor"
        }
    )

    # PyPI fetch always flows to supervisor
    graph.add_edge("fetch_pypi", "supervisor")

    # Supervisor routes to specialists (LLM-driven routing)
    graph.add_conditional_edges(
        "supervisor",
        route_specialists,
        {
            "import_specialist": "import_specialist",
            "method_specialist": "method_specialist"
        }
    )

    # Both specialists flow to report generation
    graph.add_edge("import_specialist", "generate_report")
    graph.add_edge("method_specialist", "generate_report")

    # Report generation ends the graph
    graph.add_edge("generate_report", END)

    return graph.compile()

# Build the compiled graph — imported by api/main.py
compiled_graph = build_graph()
```

```python
from agent.schemas import AgentState, ValidationReport, ValidationIssue

CONFIDENCE_THRESHOLD = 0.75

def generate_report_node(state: AgentState) -> AgentState:
    """
    Aggregates findings from all specialist agents.
    Deduplicates, applies confidence threshold, builds final report.
    """
    # Collect all findings from all specialists
    all_findings = []
    all_findings.extend(state.get("import_specialist_findings", []))
    all_findings.extend(state.get("method_specialist_findings", []))

    # Parse into ValidationIssue objects if they are dicts
    parsed_issues = []
    for finding in all_findings:
        if isinstance(finding, dict):
            try:
                parsed_issues.append(ValidationIssue(**finding))
            except Exception:
                continue
        elif isinstance(finding, ValidationIssue):
            parsed_issues.append(finding)

    # Apply confidence threshold
    confident_issues = [
        issue for issue in parsed_issues
        if issue.confidence >= CONFIDENCE_THRESHOLD
    ]

    # Deduplicate: same library + method = same issue
    seen = set()
    deduplicated = []
    for issue in confident_issues:
        key = f"{issue.library}:{issue.method}:{issue.line_number}"
        if key not in seen:
            seen.add(key)
            deduplicated.append(issue)

    report = ValidationReport(
        issues=deduplicated,
        total_issues=len(deduplicated),
        libraries_checked=state.get("libraries_checked", []),
        filtered_count=len(parsed_issues) - len(deduplicated)
    )

    return {**state, "report": report}
```
